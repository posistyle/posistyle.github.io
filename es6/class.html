    # 1.class可以看作一个语法糖，类的数据类型就是函数，类本身就指向构造函数

    class Point {
  constructor(x, y) {
    this.x = x
    this.y = y
  }

  toString() {
    return `(${this.x}, ${this.y})`
  }
}

let point = new Point(1, 2)
point, point.toString() // '(1, 2)'
typeof Point //function
Point.prototype.constructor === Point // true

    # 2.constructor -->通过new命令生成对象实例时，自动调用该方法（类必须有constructor方法，没有会默认添加）

    • 注意点

    1.constructor默认返回实例对象

    2.类必须使用New调用

    # 3.类的实例

    let point1 = new Point(3, 4)
point1.hasOwnProperty('x') // true
point1.hasOwnProperty('toString') // false
Object.getPrototypeOf(point1).hasOwnProperty('toString')// true
point1.__proto__.hasOwnProperty('toString') // true
point1.__proto__ === Point.prototype// true

    # 4.取值函数（getter）和存值函数（setter）

    class MyClass {
  constructor () {}

  get porp () {
    return 'getter'
  }
  set porp (val) {
    console.log('setter' + val)
  }
}

let inst = new MyClass()
inst.porp // 'getter'
inst.porp = 123  //'setter123'

    # 8.静态方法：加上static关键字，就表示该方法不会被实例继承，而是直接通过类来调用(如果静态方法包含this关键字，这个this指的是类，而不是实例)

    class Foo {
  constructor (name) {
    this.name = name
  }
  static classMethod () {
    console.log('hello')
    return 'hello'
  }
  method () {
    console.log('no static')
  }
}

let foo = new Foo()
Foo.classMethod() //hello
foo.method() //no static

//子类继承父类的静态方法
class Bar extends Foo {
  constructor (name, age) {
    super(name)
    this.age = age
  }
}
Bar.classMethod()

//重写父类静态方法
class Bar1 extends Foo {
  static classMethod () {
    console.log(super.classMethod() + ',too')
  }
}
Bar1.classMethod()
